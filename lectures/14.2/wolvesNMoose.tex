\section*{Associations}
\begin{description}
\item[Association]~\\
  Host 'kender til' gæst
\item[Aggregation]~\\
  Host 'har kopi af' gæst
\item[Composition]~\\
  Host 'opretter afhængig og har kopi af' gæst. Når host slettes slettes gæst ligeså.
\end{description}
