\begin{defproblem}{diffcos}
  \begin{onlyproblem}
    Differentiate $f(x)=\cos(4x)$.
  \end{onlyproblem}
  \begin{onlysolution}%
    \begin{solution}
      $f'(x) = -4\sin(4x)$
    \end{solution}
  \end{onlysolution}
\end{defproblem}

  \end{onlyproblem}
